def int calculate_sum_and_product(param int x, param int y)
{
    let int sum_val;
    let int prod_val;
    
    sum_val = x + y; 
    prod_val = x * y % (sum_val);

    return prod_val; 
};

def string concatenate_if(param string s1, param string s2, param bool condition)
{
    let string result;
    
    if(condition):
    {
        result = s1; 
    }
    else
    {
        result = s2;
    }
    return result; 
};
